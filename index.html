Supabase Table SQL
Run this once in your Supabase SQL editor:
-- 1. Create the table create table if not exists dashboard_sync ( id uuid primary key default gen_random_uuid(), user_id text not null, key text not null, value jsonb, updated_at timestamptz default now(), unique (user_id, key) ); -- 2. Grant full access to the anon key (personal app, no auth needed) grant select, insert, update, delete on dashboard_sync to anon; grant select, insert, update, delete on dashboard_sync to authenticated; -- 3. Disable RLS — simplest for a single-user personal dashboard alter table dashboard_sync disable row level security; -- 4. Auto-update timestamp on every write create or replace function set_updated_at() returns trigger language plpgsql as $$ begin new.updated_at = now(); return new; end; $$; drop trigger if exists trg_updated_at on dashboard_sync; create trigger trg_updated_at before update on dashboard_sync for each row execute procedure set_updated_at();
⚠️ If you previously ran the old SQL, also run:
alter table dashboard_sync disable row level security;
